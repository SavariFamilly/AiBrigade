# AIBrigade Test Commands - Minecraft 1.20.1
# Copier-coller ces commandes dans le jeu pour tester les fonctionnalités

# ============================================
# TESTS DE BASE - SPAWN
# ============================================

# Test 1: Spawn d'un bot solo
/aibrigade spawn solo TestBot1

# Test 2: Spawn à une position spécifique
/aibrigade spawn solo Guard1 0 70 0

# Test 3: Spawn d'un petit groupe
/aibrigade spawn group Alpha 5

# Test 4: Spawn d'un groupe moyen
/aibrigade spawn group Beta 20

# Test 5: Spawn d'un grand groupe (performance test)
/aibrigade spawn group Gamma 50

# ============================================
# TESTS DE NOMS (Si setpreset implémenté)
# ============================================

# Test 6: Preset realistic
/aibrigade setpreset TestBot1 preset:realistic

# Test 7: Preset gamer
/aibrigade setpreset Alpha preset:gamer

# Test 8: Preset humor
/aibrigade setpreset Beta preset:humor

# Test 9: Preset randomize
/aibrigade setpreset Gamma preset:randomize

# Test 10: Preset mixed
/aibrigade setpreset TestBot1 preset:mixed

# ============================================
# TESTS D'ÉQUIPEMENT
# ============================================

# Test 11: Armure diamant complète
/aibrigade givearmor TestBot1 diamond

# Test 12: Armure fer pour groupe
/aibrigade givearmor Alpha iron

# Test 13: Armure aléatoire
/aibrigade givearmor Beta random

# Test 14: Armure mixte
/aibrigade givearmor Gamma mixed

# Test 15: Armure netherite
/aibrigade givearmor Guard1 netherite

# ============================================
# TESTS DE LEADERSHIP
# ============================================

# Test 16: Assigner un leader simple

/aibrigade spawn solo Follower1

# Test 17: Chain de leadership
/aibrigade spawn solo ChiefLeader
/aibrigade spawn solo SubLeader1
/aibrigade spawn solo SubLeader2
/aibrigade assignleader SubLeader1 ChiefLeader
/aibrigade assignleader SubLeader2 SubLeader1

# Test 18: Leader de groupe
/aibrigade spawn group Squad1 10
/aibrigade spawn solo SquadLeader
/aibrigade assignleader Squad1 SquadLeader

# TESTS DE COMPORTEMENT

# Test 39: Activer targets

# Test 40: Activer ranges

# Test 41: Info du bot proche

# Test 42: Désactiver debug

# TESTS DE SUPPRESSION

/aibrigade removebot TestBot1

# ============================================
# SCÉNARIOS COMPLEXES
# ============================================

# Scénario 1: Base défensive avec gardes statiques
/aibrigade spawn group BaseGuards 4 0 70 0
/aibrigade givearmor BaseGuards diamond
/aibrigade setbehavior BaseGuards guard
/aibrigade togglestatic BaseGuards

# Scénario 2: Escouade mobile avec leader
/aibrigade spawn solo SquadCaptain 100 64 100
/aibrigade spawn group SquadMembers 10 100 64 100
/aibrigade givearmor SquadCaptain netherite
/aibrigade givearmor SquadMembers iron
/aibrigade assignleader SquadMembers SquadCaptain
/aibrigade setbehavior SquadMembers follow
/aibrigade setradius SquadMembers 12

# Scénario 3: Patrouille automatique
/aibrigade spawn group Patrol 8 -50 70 -50
/aibrigade givearmor Patrol mixed
/aibrigade setbehavior Patrol patrol

# Scénario 4: Combat entre groupes
/aibrigade spawn group AttackForce 15 -100 64 0
/aibrigade spawn group DefenseForce 15 100 64 0
/aibrigade givearmor AttackForce diamond
/aibrigade givearmor DefenseForce netherite
/aibrigade hostile AttackForce true
/aibrigade hostile DefenseForce true
